  <script>
    // --- 1. CONFIGURATION ---
    const ASSETS = {
      orb: "https://assets.zyrosite.com/A3QlOV94WLSaL6xE/shot-panoramic-composition-library-vfOqabRkgVEFwqi9.jpg",
      bg: "https://assets.zyrosite.com/A3QlOV94WLSaL6xE/hold9998_cobden-st_in12_double-height-penthouse-void-tastTN8YGC6w7PC2.jpg",
      // Using a standard Polyhaven HDR for environment reflections
      env: "https://dl.polyhaven.org/file/ph-assets/HDRIs/hdr/1k/potsdamer_platz_1k.hdr"
    };

    // --- 2. UI UTILITIES ---

    // Function to split text into spans for animation
    function setupSplitText() {
      const h1 = document.getElementById('split-target');
      // Helper to process a text node or element
      const processNode = (node) => {
        if (node.nodeType === 3) { // Text node
          const text = node.textContent;
          const span = document.createElement('span');
          span.innerHTML = text.split('').map(char => 
            `<span class="char" style="opacity:0; transform:translateX(-20px); filter:blur(10px)">${char}</span>`
          ).join('');
          node.replaceWith(span);
        } else if (node.nodeType === 1) { // Element
          Array.from(node.childNodes).forEach(processNode);
        }
      };
      Array.from(h1.childNodes).forEach(processNode);
    }

    // --- 3. THREE.JS SCENE SETUP ---
    
    // Globals
    let scene, camera, renderer;
    let bgMesh, orbGroup, innerOrb, outerOrb;
    let menuOpen = false;
    
    const container = document.getElementById('canvas-container');
    const loadingBar = document.getElementById('progress-bar');
    const loaderOverlay = document.getElementById('loader');

    function init() {
      // Scene
      scene = new THREE.Scene();
      
      // Camera
      camera = new THREE.PerspectiveCamera(35, window.innerWidth / window.innerHeight, 0.1, 100);
      camera.position.set(0, 0, 10.5);

      // Renderer
      renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
      renderer.setSize(window.innerWidth, window.innerHeight);
      renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
      renderer.toneMapping = THREE.ACESFilmicToneMapping;
      renderer.outputEncoding = THREE.sRGBEncoding;
      container.appendChild(renderer.domElement);

      // Lights (Standard Setup)
      const ambient = new THREE.AmbientLight(0xffffff, 1);
      scene.add(ambient);
      
      const spot1 = new THREE.SpotLight(0xffffff, 2);
      spot1.position.set(10, 10, 10);
      spot1.angle = 0.2;
      spot1.penumbra = 1;
      scene.add(spot1);

      const spot2 = new THREE.SpotLight(0xffffff, 2);
      spot2.position.set(-10, 10, -5);
      spot2.angle = 0.2;
      spot2.penumbra = 1;
      scene.add(spot2);

      // Start Loading Process
      loadAssets();
      
      // Events
      window.addEventListener('resize', onWindowResize);
      setupUIInteractions();
    }

    // --- 4. ASSET LOADING ---
    
    function loadAssets() {
      const manager = new THREE.LoadingManager();
      const textureLoader = new THREE.TextureLoader(manager);
      const rgbeLoader = new THREE.RGBELoader(manager);

      manager.onProgress = (url, itemsLoaded, itemsTotal) => {
        const progress = (itemsLoaded / itemsTotal) * 100;
        loadingBar.style.width = progress + '%';
      };

      manager.onLoad = () => {
        // Hide loader
        setTimeout(() => {
          loaderOverlay.style.opacity = '0';
          revealScene();
        }, 500);
        setTimeout(() => {
          loaderOverlay.style.display = 'none';
        }, 1300);
      };

      // Load Textures
      const bgTexture = textureLoader.load(ASSETS.bg);
      const orbTexture = textureLoader.load(ASSETS.orb);
      
      // Load HDR Environment
      rgbeLoader.setDataType(THREE.UnsignedByteType).load(ASSETS.env, (texture) => {
        texture.mapping = THREE.EquirectangularReflectionMapping;
        scene.environment = texture; // Apply to scene for reflections
        
        // Now create objects since assets are ready
        createObjects(bgTexture, orbTexture);
        animate();
      });
    }

    function createObjects(bgTexture, orbTexture) {
      // 1. Background Plane
      // We calculate size to fit frustum at z = -10
      bgTexture.encoding = THREE.sRGBEncoding;
      const geometry = new THREE.PlaneGeometry(1, 1);
      const material = new THREE.MeshBasicMaterial({ map: bgTexture, transparent: true });
      bgMesh = new THREE.Mesh(geometry, material);
      bgMesh.position.z = -10;
      updateBackgroundScale(); // Fit to screen
      scene.add(bgMesh);

      // 2. Orb Group
      orbGroup = new THREE.Group();
      orbGroup.position.set(3.5, 0, 0); // Position right
      scene.add(orbGroup);

      // 3. Inner Orb (The Picture)
      orbTexture.encoding = THREE.sRGBEncoding;
      const innerGeo = new THREE.SphereGeometry(1, 64, 64);
      const innerMat = new THREE.MeshBasicMaterial({ map: orbTexture });
      innerOrb = new THREE.Mesh(innerGeo, innerMat);
      innerOrb.scale.set(0, 0, 0); // Start hidden
      orbGroup.add(innerOrb);

      // 4. Outer Orb (The Glass)
      const outerGeo = new THREE.SphereGeometry(1, 64, 64);
      // NATIVE THREE.JS GLASS MATERIAL
      const outerMat = new THREE.MeshPhysicalMaterial({
        color: 0xffffff,
        transmission: 1.0,  // Glass-like transparency
        opacity: 1.0,
        metalness: 0.0,
        roughness: 0.0,
        ior: 1.5,           // Index of Refraction
        thickness: 1.5,     // Volume simulation
        specularIntensity: 1.0,
        clearcoat: 1.0,
        envMapIntensity: 1.0,
        transparent: true
      });
      
      outerOrb = new THREE.Mesh(outerGeo, outerMat);
      outerOrb.scale.set(0, 0, 0); // Start hidden
      orbGroup.add(outerOrb);
    }

    // --- 5. ANIMATION & LOGIC ---

    function revealScene() {
      // 1. Reveal UI
      const tl = gsap.timeline();
      
      tl.to(".menu-btn", { y: 0, opacity: 1, duration: 1, ease: "power3.out" });
      tl.to(".tagline", { opacity: 1, duration: 1, ease: "power2.out" }, "-=0.5");
      
      // Reveal Split Text
      const chars = document.querySelectorAll('.char');
      tl.to(chars, { 
        x: 0, 
        opacity: 1, 
        filter: "blur(0px)", 
        duration: 1.2, 
        stagger: 0.03, 
        ease: "power2.out" 
      }, "-=0.8");

      tl.to(".glass-card", { y: 0, opacity: 1, duration: 1, ease: "power3.out" }, "-=0.5");

      // 2. Reveal 3D Orb (Scale Up)
      // Using GSAP to animate Three.js properties
      gsap.to(innerOrb.scale, { x: 2.7, y: 2.7, z: 2.7, duration: 2, ease: "power3.out" });
      gsap.to(outerOrb.scale, { x: 2.71, y: 2.71, z: 2.71, duration: 2, ease: "power3.out" });
    }

    const clock = new THREE.Clock();

    function animate() {
      requestAnimationFrame(animate);
      
      const delta = clock.getDelta();
      // const elapsed = clock.getElapsedTime();

      // Float Animation
      if (orbGroup) {
        // orbGroup.position.y = Math.sin(elapsed * 2) * 0.1; // Bobbing removed
        
        // Rotation
        innerOrb.rotation.y += delta * 0.2;
        outerOrb.rotation.y += delta * 0.2;
      }

      renderer.render(scene, camera);
    }

    // --- 6. INTERACTION LOGIC ---

    function setupUIInteractions() {
      setupSplitText();

      const menuBtn = document.getElementById('menu-btn');
      const menuOverlay = document.querySelector('.menu-overlay');
      const menuItems = document.querySelectorAll('.menu-item-link');
      const menuFooter = document.querySelector('.menu-footer-right');
      const heroContent = document.getElementById('hero-content');
      const closeLinks = document.querySelectorAll('.menu-item-link');

      const toggleMenu = () => {
        menuOpen = !menuOpen;
        menuBtn.innerText = menuOpen ? "Close" : "Menu";

        if (menuOpen) {
          // Open Menu
          const tl = gsap.timeline();
          tl.to(menuOverlay, { right: 0, duration: 1.2, ease: "expo.inOut" });
          tl.to(menuItems, { opacity: 1, x: 0, duration: 1, stagger: 0.08, ease: "expo.out" }, "-=0.6");
          tl.to(menuFooter, { opacity: 1, y: 0, duration: 1, ease: "power3.out" }, "-=0.8");

          // Blur DOM Content
          gsap.to(heroContent, { opacity: 0, filter: "blur(20px)", scale: 0.9, duration: 1, ease: "power2.inOut" });

          // 3D Background Zoom & Blur Effect
          updateBackgroundScale(true); // Trigger zoom
          gsap.to(bgMesh.material, { opacity: 0.3, duration: 1.5 });

        } else {
          // Close Menu
          const tl = gsap.timeline();
          tl.to(menuItems, { opacity: 0, x: 50, duration: 0.6, stagger: 0.05, ease: "power2.in" });
          tl.to(menuFooter, { opacity: 0, y: 20, duration: 0.6 }, "-=0.6");
          tl.to(menuOverlay, { right: "-100%", duration: 1, ease: "expo.inOut" }, "-=0.4");

          // Unblur DOM Content
          gsap.to(heroContent, { opacity: 1, filter: "blur(0px)", scale: 1, duration: 1, ease: "power2.out" });

          // 3D Background Reset
          updateBackgroundScale(false); // Trigger zoom out
          gsap.to(bgMesh.material, { opacity: 1, duration: 1.5 });
        }
      };

      menuBtn.addEventListener('click', toggleMenu);
      closeLinks.forEach(link => link.addEventListener('click', () => { if(menuOpen) toggleMenu(); }));
    }

    // --- 7. RESPONSIVE UTILS ---

    function updateBackgroundScale(isZoomed = false) {
      if (!bgMesh || !camera) return;

      const distance = 20.5; // Distance of plane from camera (10.5 - (-10))
      const vFov = camera.fov * Math.PI / 180;
      const height = 2 * Math.tan(vFov / 2) * distance;
      const width = height * (window.innerWidth / window.innerHeight);

      // If menu is open (isZoomed), we add 10% scale
      const scaleMult = isZoomed ? 1.1 : 1.0;

      gsap.to(bgMesh.scale, {
        x: width * scaleMult,
        y: height * scaleMult,
        duration: 1.5,
        ease: "power2.inOut"
      });
    }

    function onWindowResize() {
      camera.aspect = window.innerWidth / window.innerHeight;
      camera.updateProjectionMatrix();
      renderer.setSize(window.innerWidth, window.innerHeight);
      updateBackgroundScale(menuOpen);
    }

    // Start App
    init();

  </script>
