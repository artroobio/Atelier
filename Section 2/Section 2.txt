    <script>
        gsap.registerPlugin(ScrollTrigger);

        // Helper to wrap words for cinematic reveal
        function wrapWords(elementId) {
            const el = document.getElementById(elementId);
            const text = el.innerHTML;
            const words = text.split(/(\s+|<br\/?>|<\/?[^>]+>)/g);
            
            el.innerHTML = words.map(word => {
                if (word.trim().length > 0 && !word.startsWith('<')) {
                    return `<span class="reveal-word">${word}</span>`;
                }
                return word;
            }).join('');
        }

        window.onload = () => {
            // Prepare cinematic elements
            wrapWords('heading');
            wrapWords('card-paragraph');

            const entryTl = gsap.timeline({
                scrollTrigger: {
                    trigger: "#philosophy",
                    start: "top 60%",
                }
            });

            // 1. Structural Line reveal
            entryTl.to("#vertical-line", {
                height: "100%",
                duration: 2,
                ease: "power4.inOut"
            });

            // 2. Tagline reveal
            entryTl.to("#tagline", {
                filter: "blur(0px)",
                opacity: 1,
                y: 0,
                duration: 1,
                ease: "power2.out"
            }, "-=1.5");

            // 3. Cinematic Heading Reveal (Word by Word)
            entryTl.to("#heading .reveal-word", {
                filter: "blur(0px)",
                opacity: 1,
                y: 0,
                duration: 1.5,
                stagger: 0.04,
                ease: "power4.out"
            }, "-=1.4");

            // 4. Glass Card Entrance
            entryTl.to("#body-card", {
                filter: "blur(0px)",
                opacity: 1,
                y: 0,
                duration: 1.2,
                ease: "power3.out"
            }, "-=1.2");

            // 5. Cinematic Paragraph Reveal (Word by Word)
            entryTl.to("#card-paragraph .reveal-word", {
                filter: "blur(0px)",
                opacity: 1,
                y: 0,
                duration: 1.2,
                stagger: 0.015,
                ease: "power3.out"
            }, "-=0.8");

            // 6. Grid items reveal
            entryTl.fromTo("#card-grid .reveal-item", 
                { opacity: 0, y: 20, filter: "blur(10px)" },
                {
                    opacity: 1,
                    y: 0,
                    filter: "blur(0px)",
                    duration: 1,
                    stagger: 0.2,
                    ease: "power2.out"
                }, 
                "-=0.6"
            );
        };
    </script>
